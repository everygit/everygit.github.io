# Submodule

# 添加子项目

git submodule add git@xxxx.xxx/xxx.git

# init

git submodule init

# update

git submodule update

# 更新子项目

cd your_submodule
git checkout master
git commit -a -m "commit in submodule"
git push
cd ..
git add your_submodule
git commit -m "Updated submodule"

注意！新 clone 的分支，需要切换到 master 分支
