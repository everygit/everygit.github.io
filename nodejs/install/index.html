# linux 源码安装

# 1 . 准备工作

yum -y install gcc make gcc-c++ openssl-devel wget

# 2. 下载源码并且解压 (NodeJs 官网)

wget https://nodejs.org/dist/v6.10.3/node-v6.10.3.tar.gz
tar -zxf node-v6.10.3.tar.gz

# 3. 编译

cd node-v6.10.3
./configure
make && make install

# 4. 验证是否安装成功

node -v
