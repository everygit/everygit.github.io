# 激活

一般在这个阶段，会清理旧版本的缓存

# 示例

self.addEventListener('activate', function (event) {
    event.waitUntil(Promise.all([
        self.clients.claim(),
        caches.keys().then(function (list) {
            return Promise.all(list.map(function (cache) {
                if (cache != ver) {
                    return caches.delete(cache);
                }
            }))
        })
    ]))
});
