# 通用类型

TypeScript 提供了几种实用工具类型来促进常见的类型转换。这些工具在整个 ts 范围可用。

# Partial<Type>

构造将 Type 的所有属性设置为可选的类型。这个实用程序将返回一个表示给定类型的所有子集的类型。

interface Todo {
  title: string;
  description: string;
}

type todo = Partial<Todo>;
// type todo = { title?: string; description?: string };

# Required<Type>

构造由 Type 设置为 required 的所有属性组成的类型。Partial的反义词

interface Todo {
  title?: string;
  description?: string;
}

type todo = Required<Todo>;
// type todo = { title: string; description: string };

# Readonly<Type>

构造一个将 Type 的所有属性设置为只读的类型，这意味着构造类型的属性不能被重新分配

interface Todo {
  title: string;
}
type todo = Readonly<Todo>;
// type todo = {
//   readonly title: string;
// };

# Record<Keys,Type>

构造一个对象类型，该对象类型的属性键是 Keys，属性值是 Type。此实用程序可用于将一个类型的属性映射到另一个类型。

interface CatInfo {
  age: number;
  breed: string;
}

type CatName = "miffy" | "boris" | "mordred";

type a = Record<CatName, CatInfo>;

// type a = {
//   miffy: CatInfo;
//   boris: CatInfo;
//   mordred: CatInfo;
// };

# Pick<Type, Keys>

从 type 中选择一组属性 Keys(字符串字面值或字符串字面值的联合)来构造类型。

interface Todo {
  title: string;
  description: string;
  completed: boolean;
}

type TodoPreview = Pick<Todo, "title" | "completed">;

// type TodoPreview = {
//   title: string;
//   completed: boolean;
// };

# Omit<Type, Keys>

通过从 type 中选择所有属性，然后删除 Keys(字符串字面值或字符串字面值的联合)来构造类型。

interface Todo {
  title: string;
  description: string;
  completed: boolean;
  createdAt: number;
}

type TodoPreview = Omit<Todo, "description">;
// type TodoPreview = {
//   title: string;
//   completed: boolean;
//   createdAt: number;
// };

# Exclude<Type, ExcludedUnion>

通过从 Type 中排除所有可分配给 ExcludedUnion 的联合成员来构造类型。

type T0 = Exclude<"a" | "b" | "c", "a">;

// type T0 = "b" | "c";
type T1 = Exclude<"a" | "b" | "c", "a" | "b">;

// type T1 = "c";
type T2 = Exclude<string | number | (() => void), Function>;

// type T2 = string | number;

# Extract<Type, Union>

通过从 Type 中提取可分配给 Union 的所有联合成员来构造类型。

type T0 = Extract<"a" | "b" | "c", "a" | "f">;

// type T0 = "a";
type T1 = Extract<string | number | (() => void), Function>;

// type T1 = () => void;

# NonNullable<Type>

通过从 Type 中排除 null 和 undefined 来构造类型

type T0 = NonNullable<string | number | undefined>;

// type T0 = string | number;
type T1 = NonNullable<string[] | null | undefined>;

// type T1 = string[];

# Parameters<Type>

从函数类型 Type 的形参中使用的类型构造元组类型。

type a = (age: number, size: number, name: string) => string;

type b = Parameters<a>;

// type b = [age: number, size: number, name: string];

不能有效的处理泛型

type a = <T>(age: number, size: T, name: string) => string;

type b = Parameters<a>;

// type b = [age: number, size: unknown, name: string];

# ConstructorParameters<Type>

从构造函数类型的类型构造元组或数组类型。它生成一个包含所有参数类型的元组类型(如果 Type 不是函数，则生成 never 类型)。

class C {
  x;
  constructor(x: number) {
    this.x = x;
  }
}

type T0 = ConstructorParameters<typeof C>;
// type T0 = [x: number];

# ReturnType<Type>

构造由函数 Type 的返回类型组成的类型。

type a = (age: number, size: number, name: string) => string;

type b = ReturnType<a>;

// type b = string;

# InstanceType<Type>

在“Type”中构造由构造函数的实例类型组成的类型。

class C {
  x = 0;
  y = 0;
}

type T0 = InstanceType<typeof C>;

// type T0 = C;
const a: T0 = { x: 10, y: 20 };

# ThisParameterType<Type>

提取函数类型的 this 参数的类型，如果函数类型没有这个参数，则提取 unknown

function toHex(this: Number) {
  return this.toString(16);
}

function numberToString(n: ThisParameterType<typeof toHex>) {
  return toHex.apply(n);
}

扩展一下

function toHex(this: Number, unit: string) {
  return this.toString(16) + unit;
}

type b = ThisParameterType<typeof toHex>;
// type b = Number

function numberToString(n: b) {
  return toHex.apply(n, ["px"]);
}

# OmitThisParameter<Type>

从 Type 中移除此 this 参数。如果 Type 没有显式声明 this 参数，则结果只是 Type。否则，将从 Type 中创建一个不带 this 参数的新函数类型。泛型被擦除，只有最后一个重载签名被传递到新的函数类型中。

function toHex(this: Number) {
  return this.toString(16);
}

const fiveToHex: OmitThisParameter<typeof toHex> = toHex.bind(5);

console.log(fiveToHex());

深入一下

function toHex(this: Number, unit: string): string {
  return this.toString(16) + unit;
}

type b = OmitThisParameter<typeof toHex>;
// type b = (unit: string) => string

const fiveToHex: b = toHex.bind(5);

console.log(fiveToHex("px"));

# ThisType<Type>

此实用程序不返回转换后的类型。相反，它充当上下文的 this 类型标记。注意，必须启用 ——noImplicitThis 标志才能使用该实用程序

type ObjectDescriptor<D, M> = {
  data?: D;
  methods?: M & ThisType<D & M>; // Type of 'this' in methods is D & M
};

function makeObject<D, M>(desc: ObjectDescriptor<D, M>): D & M {
  let data: object = desc.data || {};
  let methods: object = desc.methods || {};
  return { ...data, ...methods } as D & M;
}

let obj = makeObject({
  data: { x: 0, y: 0 },
  methods: {
    moveBy(dx: number, dy: number) {
      this.x += dx; // Strongly typed this
      this.y += dy; // Strongly typed this
    },
  },
});
